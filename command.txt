//添加文件
git add
//提交文件
git commit -m ""
//查看当前状态
git status
//查看版本差异
git diff file.txt
//回退前一个版本
git reset --hard HEAD^
//回退指定版本
git reset --hard afa324324
//查看历史记录，以便确定要回退到哪个版本。
git log
//查看命令历史，以便确定要回到未来的哪个版本。
git reflog
//查看文本
cat file.txt
//就是让这个文件回到最近一次git commit或git add时的状态。
git checkout -- file
